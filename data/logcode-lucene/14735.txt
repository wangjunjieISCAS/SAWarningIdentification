+      final boolean binary = (bits & DefaultFieldsWriter.FIELD_IS_BINARY) != 0;
+      final int numeric = bits & DefaultFieldsWriter.FIELD_IS_NUMERIC_MASK;

+      final boolean doStop;
+      if (binary) {
